PRO01.sparql
PRO01: ¿Qué campañas comerciales se han realizado en un municipio?
PREFIX edintpro: <http://vocab.linkeddata.es/datosabiertos/def/comercio/promocion#>
PREFIX esadm:    <http://vocab.linkeddata.es/datosabiertos/def/sector-publico/territorio#>
PREFIX dcterms:  <http://purl.org/dc/terms/>

SELECT DISTINCT ?municipio ?campaign ?campaignId
WHERE {
  ?campaign a edintpro:CommercialCampaign ;
            edintpro:locatedInZone ?zone ;
            dcterms:identifier ?campaignId .

  ?zone esadm:municipio ?municipio .
}
ORDER BY ?municipio ?campaignId
PRO02.sparql
PRO02: ¿Qué tipo de campañas (bonos, descuentos, dinamización) son más frecuentes?
PREFIX edintpro: <http://vocab.linkeddata.es/datosabiertos/def/comercio/promocion#>
PREFIX skos:     <http://www.w3.org/2004/02/skos/core#>

SELECT ?campaignType ?campaignTypeLabel (COUNT(?campaign) AS ?numCampaigns)
WHERE {
  ?campaign a edintpro:CommercialCampaign ;
            edintpro:hasCommercialCampaignType ?campaignType .
# En un municipio concreto
# ?campaign edintpro:locatedInZone ?zone .
# ?zone esadm:municipio <URI_DEL_MUNICIPIO> .

  OPTIONAL {
    ?campaignType skos:prefLabel ?campaignTypeLabel .
    FILTER (lang(?campaignTypeLabel) = "es")
  }
}
GROUP BY ?campaignType ?campaignTypeLabel
ORDER BY DESC(?numCampaigns)
PRO03.sparql
PRO03: ¿Cuántos comercios se han adherido a cada campaña?
PREFIX edintpro: <http://vocab.linkeddata.es/datosabiertos/def/comercio/promocion#>
PREFIX dcterms:  <http://purl.org/dc/terms/>

SELECT ?campaign ?campaignId ?numCommercialSpaces
WHERE {
  ?campaign a edintpro:CommercialCampaign ;
            dcterms:identifier ?campaignId ;
            edintpro:numberOfCommercialSpaces ?numCommercialSpaces .
}
ORDER BY DESC(?numCommercialSpaces)


# Alternativa contnado los comercios que incluye cada campaña


SELECT ?campaign ?campaignId (
      COALESCE(COUNT(DISTINCT ?commercialSpace), 0) AS ?numCommercialSpaces)  # Para considerar campañas sin comercios adheridos
WHERE {
  ?campaign a edintpro:CommercialCampaign ;
            dcterms:identifier ?campaignId .

  OPTIONAL {
    ?campaign edintpro:includes ?commercialSpace .
  }
}
GROUP BY ?campaign ?campaignId
ORDER BY DESC(?numCommercialSpaces)

PRO04: ¿Qué eventos comerciales se han celebrado y en qué ubicaciones de un municipio?
PREFIX edintpro: <http://vocab.linkeddata.es/datosabiertos/def/comercio/promocion#>
PREFIX schema:   <http://schema.org/>
PREFIX esadm:    <http://vocab.linkeddata.es/datosabiertos/def/sector-publico/territorio#>
PREFIX dcterms:  <http://purl.org/dc/terms/>

SELECT DISTINCT ?event ?eventId ?zone ?zoneLabel
WHERE {
  ?event a edintpro:CommercialEvent ;
         dcterms:identifier ?eventId ;
         edintpro:locatedInZone ?zone .

  ?zone esadm:municipio <URI_DEL_MUNICIPIO> .

  OPTIONAL {
    ?zone rdfs:label ?zoneLabel .
    FILTER (lang(?zoneLabel) = "es")
  }
}
ORDER BY ?eventId



# Variante con coordenadas de zona de ubicación

PREFIX edintpro: <http://vocab.linkeddata.es/datosabiertos/def/comercio/promocion#>
PREFIX esadm:    <http://vocab.linkeddata.es/datosabiertos/def/sector-publico/territorio#>
PREFIX geo:      <http://www.opengis.net/ont/geosparql#>
PREFIX sf:       <http://www.opengis.net/ont/sf#>
PREFIX geocore:  <https://datos.ign.es/def/geo_core#>
PREFIX dcterms:  <http://purl.org/dc/terms/>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>

SELECT DISTINCT
  ?event
  ?eventId
  ?zone
  ?zoneLabel
  ?x
  ?y
WHERE {
  # Evento comercial
  ?event a edintpro:CommercialEvent ;
         dcterms:identifier ?eventId ;
         edintpro:locatedInZone ?zone .

  # Ubicación dentro del municipio
  ?zone esadm:municipio <URI_DEL_MUNICIPIO> ;
            geo:hasGeometry ?geom .

  # Geometría puntual con coordenadas
  ?geom a sf:Point ;
        geocore:xETRS89 ?x ;
        geocore:yETRS89 ?y .

  # Etiqueta opcional de la ubicación
  OPTIONAL {
    ?zone rdfs:label ?zoneLabel .
    FILTER (lang(?zoneLabel) = "es")
  }
}
ORDER BY ?eventId
PRO05.sparql
PRO05: ¿Qué periodos del año concentran mayor número de eventos comerciales?
PREFIX edintpro: <http://vocab.linkeddata.es/datosabiertos/def/comercio/promocion#>
PREFIX schema:   <http://schema.org/>
PREFIX xsd:      <http://www.w3.org/2001/XMLSchema#>

SELECT ?month (COUNT(?event) AS ?numEvents)
WHERE {
  ?event a edintpro:CommercialEvent ;
         schema:startDate ?startDate .

  BIND(month(xsd:dateTime(?startDate)) AS ?month)
}
GROUP BY ?month
ORDER BY DESC(?numEvents)


# Variante por año y mes

PREFIX edintpro: <http://vocab.linkeddata.es/datosabiertos/def/comercio/promocion#>
PREFIX schema:   <http://schema.org/>
PREFIX xsd:      <http://www.w3.org/2001/XMLSchema#>

SELECT ?yearMonth (COUNT(?event) AS ?numEvents)
WHERE {
  ?event a edintpro:CommercialEvent ;
         schema:startDate ?startDate .

  BIND(
    CONCAT(
      STR(year(xsd:dateTime(?startDate))), "-",
      IF(month(xsd:dateTime(?startDate)) < 10,
         CONCAT("0", STR(month(xsd:dateTime(?startDate)))),
         STR(month(xsd:dateTime(?startDate))))
    ) AS ?yearMonth
  )
}
GROUP BY ?yearMonth
ORDER BY DESC(?numEvents)
PRO07.sparql
PRO06: ¿Qué zonas concentran mayor número de eventos o campañas?
PREFIX edintpro: <http://vocab.linkeddata.es/datosabiertos/def/comercio/promocion#>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?zone ?zoneLabel (COUNT(?activity) AS ?numActivities)
WHERE {
  ?activity edintpro:locatedInZone ?zone .
  
  FILTER (
    EXISTS { ?activity a edintpro:CommercialEvent } ||
    EXISTS { ?activity a edintpro:CommercialCampaign }
  )

  OPTIONAL {
    ?zone rdfs:label ?zoneLabel .
    FILTER (lang(?zoneLabel) = "es")
  }
}
GROUP BY ?zone ?zoneLabel
ORDER BY DESC(?numActivities)



# Para diferenciar entre eventos y campañasPREFIX edintpro: <http://vocab.linkeddata.es/datosabiertos/def/comercio/promocion#>

PREFIX edintpro: <http://vocab.linkeddata.es/datosabiertos/def/comercio/promocion#>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>

SELECT
   ?zone
  (SUM(?numEvents) AS ?totalEvents)
  (SUM(?numCampaigns) AS ?totalCampaigns)
WHERE {
  {
    SELECT ?zone (COUNT(?event) AS ?numEvents) (0 AS ?numCampaigns)
    WHERE {
      ?event a edintpro:CommercialEvent ;
             edintpro:locatedInZone ?zone .
    }
    GROUP BY ?zone
  }
  UNION
  {
    SELECT ?zone (0 AS ?numEvents) (COUNT(?campaign) AS ?numCampaigns)
    WHERE {
      ?campaign a edintpro:CommercialCampaign ;
                edintpro:locatedInZone ?zone .
    }
    GROUP BY ?zone
  }
}
GROUP BY ?zone
ORDER BY DESC(?totalEvents + ?totalCampaigns)
